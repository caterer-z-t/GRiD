1000 Genomes Full Pipeline Example¶

This example runs the complete GRiD pipeline (from raw CRAMs through diploid copy number estimation) using the publicly available 1000 Genomes Project high-coverage WGS dataset. No login or data access agreement is required.

  1#!/bin/bash
  2# =============================================================================
  3# GRiD — 1000 Genomes Project Example
  4# LPA KIV-2 diploid copy number estimation (hg38, chr6)
  5#
  6# Uses publicly available 30x WGS CRAMs from the 1000G high-coverage dataset.
  7# CRAMs are streamed remotely — only the LPA region is pulled per sample,
  8# so this does not require downloading 15 GB files.
  9#
 10# Usage:
 11#   bash 1000G_example.sh                  # run all samples (2504)
 12#   bash 1000G_example.sh --pop EUR        # filter to a superpopulation
 13#   bash 1000G_example.sh --n 50           # limit to first N samples
 14#   bash 1000G_example.sh --pop EUR --n 50
 15#
 16# Runtime: ~30 min for 100 samples on 8 threads (streaming-bound)
 17# =============================================================================
 18
 19set -euo pipefail
 20
 21# -----------------------------------------------------------------------
 22# Parse arguments
 23# -----------------------------------------------------------------------
 24POP_FILTER=""
 25N_SAMPLES=0   # 0 = all
 26
 27while [[ $# -gt 0 ]]; do
 28    case $1 in
 29        --pop) POP_FILTER="$2"; shift 2 ;;
 30        --n)   N_SAMPLES="$2";  shift 2 ;;
 31        *) echo "Unknown argument: $1"; exit 1 ;;
 32    esac
 33done
 34
 35# -----------------------------------------------------------------------
 36# Dependency check
 37# -----------------------------------------------------------------------
 38MISSING=()
 39for cmd in samtools mosdepth conda awk wget; do
 40    command -v "$cmd" &>/dev/null || MISSING+=("$cmd")
 41done
 42if [[ ${#MISSING[@]} -gt 0 ]]; then
 43    echo "ERROR: missing required tools: ${MISSING[*]}"
 44    echo "Install with: conda install -c bioconda ${MISSING[*]}"
 45    exit 1
 46fi
 47
 48# -----------------------------------------------------------------------
 49# Paths and parameters
 50# -----------------------------------------------------------------------
 51SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
 52GRID_ROOT="$(dirname "$SCRIPT_DIR")"                      # repo root
 53
 54WORK_DIR="${WORK_DIR:-$(pwd)/1000G_grid_run}"             # override with env var
 55CRAM_DIR="$WORK_DIR/crams"
 56LOG_DIR="$WORK_DIR/logs"
 57DATA_DIR="$WORK_DIR/data"
 58OUTPUT_DIR="$WORK_DIR/output"
 59MOSDEPTH_WORK="$WORK_DIR/mosdepth_work"
 60
 61THREADS="${SLURM_CPUS_PER_TASK:-$(nproc)}"                # auto-detect cores
 62STREAM_JOBS=$(( THREADS > 4 ? 4 : THREADS ))              # parallel CRAM streams
 63
 64# -----------------------------------------------------------------------
 65# 1000 Genomes public URLs (no login required)
 66# -----------------------------------------------------------------------
 67BASE_URL="https://ftp.1000genomes.ebi.ac.uk/vol1/ftp"
 68CRAM_BASE="${BASE_URL}/data_collections/1000G_2504_high_coverage/working/20190425_NYGC_GATK"
 69PANEL_URL="${BASE_URL}/release/20130502/integrated_call_samples_v3.20200731.ALL.panel"
 70REF_URL="https://ftp.ncbi.nlm.nih.gov/genomes/all/GCA/000/001/405/GCA_000001405.15_GRCh38/seqs_for_alignment_pipelines.ucsc_ids/GCA_000001405.15_GRCh38_no_alt_analysis_set.fna.gz"
 71
 72# VNTR region of interest — read from bundled regions file
 73REGIONS_FILE="$GRID_ROOT/files/734_possible_coding_vntr_regions.IBD2R_gt_0.25.uniq.txt"
 74read -r CHR START END < <(awk '$7=="LPA" {print $1, $2, $3; exit}' "$REGIONS_FILE")
 75REGION="chr${CHR}:${START}-${END}"
 76
 77REPEAT_MASK="$GRID_ROOT/files/repeat_mask_list.hg38.ucsc_bed"
 78
 79# -----------------------------------------------------------------------
 80# Setup
 81# -----------------------------------------------------------------------
 82mkdir -p "$CRAM_DIR" "$LOG_DIR" "$DATA_DIR" "$OUTPUT_DIR" "$MOSDEPTH_WORK"
 83
 84timestamp() { date '+%Y-%m-%d %H:%M:%S'; }
 85log() { echo "[$(timestamp)] $*" | tee -a "$LOG_DIR/pipeline.log"; }
 86
 87log "================================================="
 88log "  GRiD — 1000 Genomes Example"
 89log "  Locus:   LPA KIV-2  ($REGION, hg38)"
 90log "  Threads: $THREADS  |  Parallel streams: $STREAM_JOBS"
 91log "  Output:  $OUTPUT_DIR"
 92log "================================================="
 93
 94# -----------------------------------------------------------------------
 95# Activate environment
 96# -----------------------------------------------------------------------
 97source "$(conda info --base)/etc/profile.d/conda.sh"
 98conda activate grid-env
 99
100# -----------------------------------------------------------------------
101# Phase 1 — Reference genome
102# -----------------------------------------------------------------------
103REF_FA="$DATA_DIR/GRCh38_no_alt.fa"
104
105if [[ ! -f "$REF_FA" ]]; then
106    log "Downloading GRCh38 reference genome..."
107    wget -q -O - "$REF_URL" | gunzip > "$REF_FA"
108    samtools faidx "$REF_FA"
109    log "Reference genome ready: $REF_FA"
110else
111    log "Reference genome already present — skipping download."
112fi
113
114# -----------------------------------------------------------------------
115# Phase 2 — Sample list
116# -----------------------------------------------------------------------
117PANEL="$DATA_DIR/1000G_panel.txt"
118SAMPLES_FILE="$DATA_DIR/samples.txt"
119
120if [[ ! -f "$PANEL" ]]; then
121    log "Downloading 1000G panel file..."
122    wget -q -O "$PANEL" "$PANEL_URL"
123fi
124
125# Columns: sample  pop  super_pop  gender
126if [[ -n "$POP_FILTER" ]]; then
127    log "Filtering to superpopulation: $POP_FILTER"
128    awk -v pop="$POP_FILTER" 'NR>1 && $3==pop {print $1}' "$PANEL" > "$SAMPLES_FILE"
129else
130    awk 'NR>1 {print $1}' "$PANEL" > "$SAMPLES_FILE"
131fi
132
133TOTAL=$(wc -l < "$SAMPLES_FILE")
134if [[ "$N_SAMPLES" -gt 0 && "$N_SAMPLES" -lt "$TOTAL" ]]; then
135    head -n "$N_SAMPLES" "$SAMPLES_FILE" > "${SAMPLES_FILE}.tmp" && mv "${SAMPLES_FILE}.tmp" "$SAMPLES_FILE"
136    log "Using first $N_SAMPLES of $TOTAL available samples."
137else
138    log "Using all $TOTAL samples."
139fi
140
141# -----------------------------------------------------------------------
142# Phase 3 — Stream LPA-region CRAMs (parallel, remote)
143#
144# Only the target region is pulled over the network per sample.
145# A 30x whole-genome CRAM is ~15 GB; the LPA region slice is ~2 MB.
146# -----------------------------------------------------------------------
147log "Streaming LPA region CRAMs from 1000G EBI..."
148
149stream_cram() {
150    local sample="$1"
151    local url="${CRAM_BASE}/${sample}.hg38.cram"
152    local out="${CRAM_DIR}/${sample}.cram"
153
154    if [[ -f "$out" && -f "${out}.crai" ]]; then
155        return 0   # already done
156    fi
157
158    samtools view -T "$REF_FA" -b "$url" "$REGION" \
159        | samtools sort -@ 2 -o "$out" && \
160    samtools index "$out"
161}
162
163export -f stream_cram
164export REF_FA CRAM_BASE CRAM_DIR REGION
165
166# Use GNU parallel if available, otherwise xargs
167if command -v parallel &>/dev/null; then
168    parallel -j "$STREAM_JOBS" --bar stream_cram :::: "$SAMPLES_FILE" \
169        >> "$LOG_DIR/stream_crams.log" 2>&1
170else
171    xargs -P "$STREAM_JOBS" -I {} bash -c 'stream_cram "$@"' _ {} \
172        < "$SAMPLES_FILE" >> "$LOG_DIR/stream_crams.log" 2>&1
173fi
174
175log "CRAM streaming complete."
176
177# -----------------------------------------------------------------------
178# Phase 4 — Generate config
179# -----------------------------------------------------------------------
180CONFIG="$WORK_DIR/config.yaml"
181
182cat > "$CONFIG" << YAML
183# Auto-generated GRiD config — 1000 Genomes LPA KIV-2 example
184samples_file: "$SAMPLES_FILE"
185directory_loc: "$CRAM_DIR"
186reference_genome: "$REF_FA"
187output_dir: "$OUTPUT_DIR"
188threads: $THREADS
189file_type: "cram"
190chrom: "chr${CHR}"
191start_bp: $START
192end_bp: $END
193output_file_type: "tsv"
194
195index:
196  run: False                     # CRAMs already indexed above
197
198count_reads:
199  run: True
200  output_file_prefix: "read_counts"
201  min_mapq: 1
202  flags:
203    - 83    # proper pair, read reverse strand
204    - 147   # proper pair, mate reverse strand
205    - 81    # read reverse strand
206    - 145   # mate reverse strand
207
208mosdepth:
209  run: True
210  output_file_prefix: "mosdepth_results"
211  bin_size: 1000
212  mode: "fast"
213  region_name: "LPA"
214  work_dir: "$MOSDEPTH_WORK"
215  remove_intermediate: True
216
217  normalize:
218    run: True
219    min_depth: 20
220    max_depth: 100
221    top_frac: 0.1
222    output_file_prefix: "mosdepth_normalized"
223    repeat_mask_file: "$REPEAT_MASK"
224
225  neighbors:
226    run: True
227    output_file_prefix: "neighbors"
228    num_neighbors: 5
229    zmax: 2.0
230    sigma2_max: 1000
231
232compute_diploid_genotypes:
233  run: True
234  output_file_prefix: "diploid_genotypes"
235
236compute_haploid_genotypes:
237  run: False                     # requires IBS file — see 1000G_IBS_example.sh
238  output_file_prefix: "haploid_genotypes"
239  ibs_output: "$DATA_DIR/ibs_neighbors_chr6.tsv.gz"
240  min_neighbors: 1
241  max_neighbors: 10
242  n_iters: 100
243YAML
244
245log "Config written to: $CONFIG"
246
247# -----------------------------------------------------------------------
248# Phase 5 — Run GRiD
249# -----------------------------------------------------------------------
250log "Running GRiD pipeline..."
251grid wgs "$CONFIG" 2>&1 | tee "$LOG_DIR/grid.log"
252
253log "================================================="
254log "  Done. Diploid CN results in: $OUTPUT_DIR"
255log "  Next: run 1000G_IBS_example.sh to generate the"
256log "  IBS neighbors file, then re-run with"
257log "  compute_haploid_genotypes.run: True"
258log "================================================="

Usage¶

Run all samples from a single superpopulation:

bash examples/1000G_example.sh --pop EUR

Limit to 50 samples for a quick test:

bash examples/1000G_example.sh --pop EUR --n 50

Submit to SLURM:

sbatch --cpus-per-task=16 --mem=32G examples/1000G_example.sh --pop EUR --n 100

Override the working directory:

WORK_DIR=/scratch/my_run bash examples/1000G_example.sh --pop AFR

Next Step — Haplotype Inference¶

The pipeline above estimates diploid copy number. To decompose into haplotype-specific copy numbers, you need an IBS neighbors file first. See IBS/IBD Neighbor Computation Example for instructions, then re-run the pipeline with compute_haploid_genotypes.run: True in the generated config.
